237 void DrawGLScene(GLvoid) // Here's Where We Do All The Drawing
238 {
239  GLfloat tx,ty,tz; // Temp X, Y & Z Variables
240  VERTEX q; // Holds Returned Calculated Values For One Vertex
241  int i=0;
242 
243  xrot+=xspeed; yrot+=yspeed; zrot+=zspeed; // Increase xrot,yrot & zrot by xspeed, yspeed & zspeed
244 
245  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT); // Clear The Screen And The Depth Buffer
246  glLoadIdentity(); // Reset The View
247  glTranslatef(cx,cy,cz); // Translate The The Current Position To Start Drawing
248  glRotatef(xrot,1,0,0); // Rotate On The X Axis By xrot
249  glRotatef(yrot,0,1,0); // Rotate On The Y Axis By yrot
250  glRotatef(zrot,0,0,1); // Rotate On The Z Axis By zrot
251 
252  glBegin(GL_POINTS); // Begin Drawing Points
253  for(i=0;i<morph1.verts;i++) // Loop Through All The Verts Of morph1 (All Objects Have
254  { // The Same Amount Of Verts For Simplicity, Could Use maxver Also)
255  if(morph) q=calculate(i); else q.x=q.y=q.z=0; // If morph Is True Calculate Movement Otherwise Movement=0
256  helper.points[i].x-=q.x; // Subtract q.x Units From helper.points[i].x (Move On X Axis)
257  helper.points[i].y-=q.y; // Subtract q.y Units From helper.points[i].y (Move On Y Axis)
258  helper.points[i].z-=q.z; // Subtract q.z Units From helper.points[i].z (Move On Z Axis)
259  tx=helper.points[i].x; // Make Temp X Variable Equal To Helper's X Variable
260  ty=helper.points[i].y; // Make Temp Y Variable Equal To Helper's Y Variable
261  tz=helper.points[i].z; // Make Temp Z Variable Equal To Helper's Z Variable
262 
263  glColor3f(0,1,1); // Set Color To A Bright Shade Of Off Blue
264  glVertex3f(tx,ty,tz); // Draw A Point At The Current Temp Values (Vertex)
265  glColor3f(0,0.5f,1); // Darken Color A Bit
266  tx-=2*q.x; ty-=2*q.y; ty-=2*q.y; // Calculate Two Positions Ahead
267  glVertex3f(tx,ty,tz); // Draw A Second Point At The Newly Calculate Position
268  glColor3f(0,0,1); // Set Color To A Very Dark Blue
269  tx-=2*q.x; ty-=2*q.y; ty-=2*q.y; // Calculate Two More Positions Ahead
270  glVertex3f(tx,ty,tz); // Draw A Third Point At The Second New Position
271  } // This Creates A Ghostly Tail As Points Move
272  glEnd(); // Done Drawing Points
273 
274  // If We're Morphing And We Haven't Gone Through All 200 Steps Increase Our Step Counter
275  // Otherwise Set Morphing To False, Make Source=Destination And Set The Step Counter Back To Zero.
276  if(morph && step<=steps)step++; else { morph=FALSE; sour=dest; step=0;}
277 }
